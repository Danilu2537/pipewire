99 static inline int pw_array_ensure_size(struct pw_array *arr, size_t size)
100 {
101  size_t alloc, need;
102 
103  alloc = arr->alloc;
104  need = arr->size + size;
105 
106  if (SPA_UNLIKELY(alloc < need)) {
107  void *data;
108  alloc = SPA_MAX(alloc, arr->extend);
109  while (alloc < need)
110  alloc *= 2;
111  if (SPA_UNLIKELY((data = realloc(arr->data, alloc)) == NULL))
112  return -errno;
113  arr->data = data;
114  arr->alloc = alloc;
115  }
116  return 0;
117 }
