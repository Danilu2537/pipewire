303 pw_core_proxy_errorv(struct pw_core_proxy *core, uint32_t id, int seq,
304  int res, const char *message, va_list args)
305 {
306  char buffer[1024];
307  vsnprintf(buffer, sizeof(buffer), message, args);
308  buffer[1023] = '\0';
309  return pw_core_proxy_error(core, id, seq, res, buffer);
310 }
